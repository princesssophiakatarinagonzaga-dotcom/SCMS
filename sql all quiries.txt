01_create_roles_table.sql
CREATE TABLE roles (
    role_id   NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    role_name VARCHAR2(50) UNIQUE NOT NULL
);


02_create_users_table.sql
CREATE TABLE users (
    user_id              NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    school_id            VARCHAR2(20)  NOT NULL,
    first_name           VARCHAR2(100) NOT NULL,
    last_name            VARCHAR2(100) NOT NULL,
    email                VARCHAR2(150) UNIQUE NOT NULL,
    password             VARCHAR2(255),           -- BCrypt hash set after OTP verify
    role_id              NUMBER        NOT NULL,
    department_id        NUMBER,                  -- FK to departments (optional, for joins)
    department           VARCHAR2(150),           -- plain text, used by servlets directly
    program              VARCHAR2(150),           -- STUDENT only
    role_label           VARCHAR2(50),            -- e.g. 'Manager', 'Staff', 'Admin'
    access_status        VARCHAR2(20)  DEFAULT 'Active' NOT NULL,
    is_verified          CHAR(1)       DEFAULT 'N' NOT NULL,
    otp_code             VARCHAR2(6),
    otp_type             VARCHAR2(20),            -- 'REGISTER' | 'LOGIN' | 'RESET'
    otp_expiry           TIMESTAMP,               -- FIX: was named temp_expiry before
    temp_password        VARCHAR2(255),
    must_change_password CHAR(1)       DEFAULT 'N',
    created_at           TIMESTAMP     DEFAULT CURRENT_TIMESTAMP NOT NULL,

    CONSTRAINT fk_users_role
        FOREIGN KEY (role_id) REFERENCES roles(role_id),

    CONSTRAINT chk_access_status
        CHECK (access_status IN ('Active','Suspended')),

    CONSTRAINT chk_is_verified
        CHECK (is_verified IN ('Y','N')),

    CONSTRAINT chk_must_change
        CHECK (must_change_password IN ('Y','N')),

    CONSTRAINT uq_school_email
        UNIQUE (school_id, email)
);


03_create_departments_table.sql
CREATE TABLE departments (
    department_id 	NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    department_name VARCHAR2(150) UNIQUE NOT NULL,
    manager_id 	NUMBER UNIQUE,
    
CONSTRAINT fk_dept_manager 
FOREIGN KEY (manager_id)
        	REFERENCES users(user_id)
ON DELETE SET NULL
);



04_create_main_categories_table.sql
CREATE TABLE main_categories (
    category_id 	NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    category_name 	VARCHAR2(100) UNIQUE NOT NULL
);


05_create_concern_types_table.sql
CREATE TABLE concern_types (
    type_id 		NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    category_id 	NUMBER NOT NULL,
    type_name 	VARCHAR2(150) NOT NULL,
    
CONSTRAINT fk_type_category 
FOREIGN KEY (category_id)
        	REFERENCES main_categories(category_id)
ON DELETE CASCADE
);



06_create_concerns_table.sql
CREATE TABLE concerns (
concern_id 		NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
student_id 		NUMBER NOT NULL,
category_id 		NUMBER NOT NULL,
type_id 		NUMBER NOT NULL,
title 			VARCHAR2(200) NOT NULL,
department_id 	NUMBER NOT NULL,
description 		CLOB NOT NULL,
attachment_path 	VARCHAR2(500),
priority 		VARCHAR2(20) DEFAULT 'Low' NOT NULL,
status 		VARCHAR2(20) DEFAULT 'Pending' NOT NULL,
assigned_manager_id NUMBER,
created_at 		TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at 		TIMESTAMP,

CONSTRAINT fk_concern_student 
FOREIGN KEY (student_id)
       	REFERENCES users(user_id)
ON DELETE CASCADE,

CONSTRAINT fk_concern_category
FOREIGN KEY (category_id)
REFERENCES main_categories(category_id),

CONSTRAINT fk_concern_type
FOREIGN KEY (type_id)
REFERENCES concern_types(type_id),

CONSTRAINT fk_concern_department
FOREIGN KEY (department_id)
REFERENCES departments(department_id),

CONSTRAINT fk_concern_manager
FOREIGN KEY (assigned_manager_id)
REFERENCES users(user_id)
ON DELETE SET NULL,

    	CONSTRAINT chk_priority 
CHECK (priority IN ('Low','Medium','High','Critical')),

    	CONSTRAINT chk_status 
CHECK (status IN ('Pending','In Progress','Closed'))
);

07_create_messages_table.sql
CREATE TABLE messages (
    message_id   NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    concern_id   NUMBER NOT NULL,
    sender_id    NUMBER NOT NULL,
    message_text CLOB NOT NULL,
    sent_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,

    	CONSTRAINT fk_msg_concern 
FOREIGN KEY (concern_id)
    		REFERENCES concerns(concern_id)
ON DELETE CASCADE,

    	CONSTRAINT fk_msg_sender
FOREIGN KEY (sender_id)
REFERENCES users(user_id)
ON DELETE CASCADE
);

08_create_activity_logs_table.sql
CREATE TABLE activity_logs (
    log_id             NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    concern_id         NUMBER NOT NULL,
    action_by          NUMBER NOT NULL,
    action_type        VARCHAR2(100) NOT NULL,
    action_description VARCHAR2(500),
    action_time        TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    
CONSTRAINT fk_log_user 
FOREIGN KEY (action_by)
REFERENCES users(user_id)
ON DELETE CASCADE,

    	CONSTRAINT fk_log_concern 
FOREIGN KEY (concern_id)
REFERENCES concerns(concern_id)
ON DELETE CASCADE
);

11_indexes_and_triggers.sql
-- INDEXES FOR FILTERING
CREATE INDEX idx_concern_status ON concerns(status);
CREATE INDEX idx_concern_priority ON concerns(priority);
CREATE INDEX idx_concern_department ON concerns(department_id);
CREATE INDEX idx_concern_manager ON concerns(assigned_manager_id);
CREATE INDEX idx_concern_student ON concerns(student_id);
CREATE INDEX idx_concern_created_at ON concerns(created_at);

CREATE INDEX idx_messages_concern ON messages(concern_id);
CREATE INDEX idx_activity_concern ON activity_logs(concern_id);

-- TRIGGER FOR AUTO-UPDATING updated_at
CREATE OR REPLACE TRIGGER trg_update_concern
BEFORE UPDATE ON concerns
FOR EACH ROW
BEGIN
    :NEW.updated_at := CURRENT_TIMESTAMP;
END;
